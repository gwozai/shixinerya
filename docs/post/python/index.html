不太会用python的数组，搞了好久没有搞出来😟

我的大概思路是：

用一个循环数组来存储数据：

当前数 == 随机数 ： 平局

当前的下一个 == 随机数 ：你输了

当前数的下一个的下一个 == 随机数 ：你赢了

def oneLine() :
    print("-" * 20)

    for i in range(a) :
        oneLine()
nLine(5)
def threeNSum(a,b,c) :
    return a + b + c
def avgThree(a,b,c) :
    return threeNSum(a,b,c) // 3
print("三个数的和：",threeNSum(10,50,40))
print("三个数的平均值",avgThree(10,50,40))

def fWrite(a,b) :
    try :
        f = open(a,"w",encoding="utf-8") # utf-8的方式打开文件
        try :
            f.write(b)
        finally:
            # print("文件已经关闭")
            f.close()
    except Exception as result :
        print(result)
def wWrite(a,b) :
    try :
        f = open(a,"r",encoding="utf-8")
        c = open(b,"w",encoding="utf-8")
        try :
            tmp = f.readlines()
            str = ""
            for i in tmp :
                str += i
            c.write(str)
        finally:
            f.close()
            c.close()
            # print("文件已经关闭")
    except Exception as result :
        print(result)
