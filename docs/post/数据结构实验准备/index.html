摘要：

题目: 学生通讯录管理系统

设计内容：

建立某班学生的通讯录，要求用链表存储。

具体功能包括：

1）可以实现插入一个同学的通讯录记录；

2）能够删除某位同学的通讯录；

3）对通讯录打印输出。

设计要求：

（1）定义通讯录内容的结构体；

（2）建立存储通讯录的链表结构并初始化；

（3）建立主函数：

​ 1）建立录入函数（返回主界面）

​ 2）建立插入函数（返回主界面）

​ 3）建立删除函数（返回主界面）

​ 4）建立输出和打印函数（返回主界面）

​ I）通过循环对所有成员记录输出

​ II）输出指定姓名的某个同学的通讯录记录

​ 5）退出

#include <stdio.h>
#include <conio.h>
#include <stdlib.h>
#include<string.h>
//#include<algorithm>
/**
清除当前页面打印新的页面
给每个操作都写一个图形界面
对某个姓名进行修改：删除和添加新的记录
去掉重复的人名并且一个人可以拥有两电话号 ,插入时判断是否有该学生信息，有的话打印学生信息 
可以使用文件进行存储，保存记录值。
*/
typedef struct _studentinfo{
	char name[50];
	int telephoneNum;
}StudentInfo;
typedef struct _node{
	StudentInfo studentInfo;
	struct _node *next;
}Node;

//初始化线性表 
Node *head = NULL;

//录入学生信息
void  initStudent(){
	StudentInfo stu[2];
	strcpy(stu[0].name,"zhangsan");
	strcpy(stu[1].name,"lisi");
	strcpy(stu[1].telephoneNum,"234323535235");
	int i;
	printf("\n已运行循环");
	for(i=0;i<2;i++){			
		Node *p = (Node*)malloc(sizeof(Node));
		p->next = NULL; 
		strcpy(p->studentInfo.name,stu[i].name);
		if(!head){
			head = p;
		}else{ 
			p->next = head;
			head = p;
		}
	}
}

//信息插入:使用头插法, 
void addStudentInfo(){
	Node *p = (Node*)malloc(sizeof(Node));
	p->next = NULL;
	printf("\n请输入学生的姓名:");
	scanf("%s",&p->studentInfo.name);
	
	//判断是否有当前学生，如果有的话就终止插入，如果想要修改电话信息，就删除当前学生信息重新插入 
	Node* q = head;
	while (q){
		if(strcmp(p->studentInfo.name,q->studentInfo.name) == 0){
			printf("\n已有该学生信息，如果想要修改手机号，请删除后重新添加。");
			printf("\n%s %d",q->studentInfo.name,q->studentInfo.telephoneNum);
			printf("\n\n");
			system("pause");
			return;

		} 
		q = p->next;
	}
	
	printf("\n请输入学生的电话:"); 
	scanf("%d",&p->studentInfo.telephoneNum);
	
	if(!head){
		head = p;
	}else{ 
		p->next = head;
		head = p;
	}
	printf("\n添加成功 -_-");
	
	printf("\n\n");
	system("pause");
}

//打印学生信息
void PrintStudentInfo(){
	//遍历链表
	Node* p = head;
	if (p == NULL){
		
		printf("\n|\t\t\t\t没有学生信息\t\t\t\t\t|\n");
		printf("\n---------------------------------------------------------------------------------\n");
		printf("\n\n");
		system("pause");
		return;
	}
	while (p){
		printf("\n%s %d",p->studentInfo.name,p->studentInfo.telephoneNum);
		p = p->next;
	}
	printf("\n\n");
	system("pause");

}

//打印电话号码，输入学生姓名可以打印对应的学生号码； 
void search(){
	printf("\n请输入要查找的学生的姓名:");
	char ch[50];
	scanf("%s",&ch);
	//遍历链表
	Node* p = head;
	while (p){
		if(strcmp(ch,p->studentInfo.name) == 0){
			printf("\n%s %d",p->studentInfo.name,p->studentInfo.telephoneNum);
			break;
		} 
		p = p->next;
	}
	if(!p){
		printf("\n##未有此学生信息##"); 
	} 
	printf("\n\n");
	system("pause");
}

//删除
void rmStudentInfo(){
	printf("\n请输入要删除的学生的姓名(取消按q):");
	char ch[50];
	scanf("%s",&ch);

	//遍历链表
	Node* p = head;
	Node* q = p;
	while (p){
		if(strcmp(ch,"q")==0){
			printf("放弃本次操作"); 
			break;
		}
		if(p=q){
			if(strcmp(ch,p->studentInfo.name) == 0){
				printf("\n已删除: %s %d",p->studentInfo.name,p->studentInfo.telephoneNum); 
				head = p->next;
				free(q);
				break;
			}
		} 
		if(strcmp(ch,p->next->studentInfo.name) == 0){
			
			
			printf("\n已删除: %s %d",p->next->studentInfo.name,p->next->studentInfo.telephoneNum); 
			if(!p->next->next){
				p->next = NULL;
			}else{
				q = p->next; //暂时存储要删除的节点 
				p->next = p->next->next; //删除 
				free(q); //释放空间 
			}
			break;
		}
		p = p->next;
	}
	if(!p){
		printf("\n##未有此学生信息##");
	}
	
	printf("\n\n");
	system("pause");

}

//主管理界面
void Interface(){
	printf("\n学生通讯录管理系统：");
	printf("\n输入1添加学生信息\n输入2打印学生信息\n输入3查找某个学生\n输入4删除的学生");
}

void management(){
	
	initStudent();
	
	while (1)
	{
		// 清除输出
		system("cls");
		Interface();
		printf("\n请输入数字：");
		char ch = _getch();
		switch (ch)
		{
		case '1':
			//插入学生信息
			addStudentInfo(); 
			break;
		case '2':
			//打印学生信息
			PrintStudentInfo();
			break;
		case '3':
			//查找指定姓名的学生
			search();
			break; 
		case '4':
			//删除学生信息 
			rmStudentInfo(); 
			break; 
		case '0':
			//退出管理系统 
			exit(0);
			break;
		default:
			break;
		}
	}
}
int main(){
	management();
}
