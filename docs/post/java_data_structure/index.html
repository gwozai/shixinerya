public class arry {
    public static void main(String args[]) {
        System.out.println("hello,world");
        /**
         * 稀疏数组
         */
        //创建一个棋盘
        int arrChese[][] = new int[11][11];
        arrChese[2][4] = 2;
        arrChese[3][5] = 1;
        arrChese[6][4] = 2;
        // 输出原始的二维数组
        int sum = 0;
        System.out.println("输出原始的二维数组：");
        for (int i = 0; i < arrChese.length; i++) {
            for (int j = 0; j < arrChese.length; j++) {
                System.out.print(arrChese[i][j] + "\t");
                if (arrChese[i][j] != 0) {
                    sum++;
                }
            }
            System.out.println();
        }
        // 改为稀疏数组
        int sparse[][] = new int[sum + 1][sum];
        sparse[0][0] = 11;
        sparse[0][1] = 11;
        sparse[0][2] = sum;
        int count = 0;
        for (int i = 1; i < arrChese.length; i++) {
            for (int j = 0; j < arrChese.length; j++) {
                if (arrChese[i][j] != 0) {
                    count++;
                    sparse[count][0] = i;
                    sparse[count][1] = j;
                    sparse[count][2] = arrChese[i][j];
                }
            }
        }

        //打印稀疏数组
        for (int i = 0; i < sum + 1; i++) {
            for (int j = 0; j < sum; j++) {
                System.out.print(sparse[i][j] + "\t");
            }
            System.out.println();
        }

        // 还原稀疏数组，创建数组
        int original[][] = new int[sparse[0][0]][sparse[0][1]];
        for (int i = 1; i < sparse.length; i++) {
            original[sparse[i][0]][sparse[i][1]] = sparse[i][2];
        }
        System.out.println("还原后的稀疏数组");
        // 打印数组数据
        for (int i = 0; i < 11; i++) {
            for (int j = 0; j < 11; j++) {
                System.out.print(original[i][j] + "\t");
            }
            System.out.println();
        }

        //如何判断二位数组的行和列长度

    }
}
